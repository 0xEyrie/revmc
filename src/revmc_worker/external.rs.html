32#[derive(Debug)]
33pub struct EvmCompilerFnTuple((EvmCompilerFn, Arc<Library>));
34/// Compiler Worker as an external context.
35///
36/// External function fetching is optimized by using an LRU Cache.
37/// In many cases, a contract that is called will likely be called again,
38/// so the cache helps reduce library loading cost.
39pub struct EXTCompileWorker {
40    worker_pool: AotCompileWorkerPool,
41    pub cache: RwLock<LruCache<B256, EvmCompilerFnTuple>>,
42}
43
44impl EXTCompileWorker {
45    pub fn new(
46        primary: bool,
47        threshold: u64,
48        worker_pool_size: usize,
49        cache_size_words: usize,
50    ) -> Result<Self, Error> {
51        let hot_code_counter = HotCodeCounter::new(primary, worker_pool_size)?;
52        let worker_pool = AotCompileWorkerPool::new(threshold, hot_code_counter, worker_pool_size);
53
54        Ok(Self {
55            worker_pool,
56            cache: RwLock::new(LruCache::new(NonZeroUsize::new(cache_size_words).unwrap())),
57        })
58    }
59
60    /// Fetches the compiled function from disk, if exists
61    pub fn get_function(&self, code_hash: &B256) -> Result<FetchedFnResult, Error> {
62        if code_hash.is_zero() {
63            return Ok(FetchedFnResult::NotFound);
64        }
65        // Write locks are required for reading from LRU Cache
66        {
67            let mut acq = true;
68
69            let cache = match self.cache.try_write() {
70                Ok(c) => Some(c),
71                Err(err) => match err {
72                    /* in this case, read from file instead of cache */
73                    TryLockError::WouldBlock => {
74                        acq = false;
75                        None
76                    }
77                    TryLockError::Poisoned(err) => Some(err.into_inner()),
78                },
79            };
80
81            if acq {
82                if let Some(t) = cache.unwrap().get(code_hash) {
83                    return Ok(FetchedFnResult::Found(t.0 .0));
84                }
85            }
86        }
87
88        let name = module_name();
89        let so = store_path().join(code_hash.to_string()).join("a.so");
90        if so.try_exists().unwrap_or(false) {
91            {
92                let lib = Arc::new((unsafe { Library::new(so) })?);
93                let f: Symbol<'_, revmc::EvmCompilerFn> = unsafe { lib.get(name.as_bytes())? };
94
95                let tuple = EvmCompilerFnTuple((*f, lib.clone()));
96                let mut cache = self
97                    .cache
98                    .write()
99                    .map_err(|err| Error::RwLockPoison { err: err.to_string() })?;
100                cache.put(*code_hash, tuple);
101
102                return Ok(FetchedFnResult::Found(*f));
103            }
104        }
105        Ok(FetchedFnResult::NotFound)
106    }
107
108    /// Spwan AOT compile the byecode referred by code_hash
109    pub fn spwan(&self, spec_id: SpecId, code_hash: B256, bytecode: Bytes) -> Result<(), Error> {
110        self.worker_pool.spwan(spec_id, code_hash, bytecode)?;
111        Ok(())
112    }
