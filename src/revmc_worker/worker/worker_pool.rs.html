17/// A worker responsible for compiling bytecode in machine code.
18#[derive(Debug)]
19pub struct AotCompileWorkerPool {
20    pub threshold: u64,
21    semaphore: Arc<Semaphore>,
22    inner: Arc<WorkerPoolInner>,
23}

25impl Debug for WorkerPoolInner {
26    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
27        f.debug_struct("WorkerPoolInner")
28            .field("hot_code_counter", &self.hot_code_counter)
29            .field("aot_compiler", &self.aot_compiler)

31    }

34struct WorkerPoolInner {
35    hot_code_counter: HotCodeCounter,
36    aot_compiler: Mutex<AotCompiler>,
37}
38
39impl AotCompileWorkerPool {
40    /// Creates a new `CompileWorkerPool`.
41    ///
42    /// # Arguments
43    ///
44    /// * `threshold` - The threshold for the number of times a bytecode must be seen before it is
45    ///   compiled.
46    /// * `hot_code_counter` - A reference-counted, thread-safe handle to count call of contract
47    /// * `max_concurrent_tasks` - The maximum number of concurrent aot compiling tasks allowed.
48    pub(crate) fn new(
49        threshold: u64,
50        hot_code_counter: HotCodeCounter,
51        max_concurrent_tasks: usize,
52    ) -> Self {
53        Self {
54            threshold,
55            semaphore: Arc::new(Semaphore::new(max_concurrent_tasks)),
56            inner: Arc::new(WorkerPoolInner {
57                hot_code_counter,
58                aot_compiler: Mutex::new(AotCompiler::new(AotConfig::default())),
59            }),
60        }
61    }
62
63    #[allow(dead_code)]
64    pub(crate) fn new_with_config(
65        threshold: u64,
66        hot_code_counter: HotCodeCounter,
67        max_concurrent_tasks: usize,
68        config: AotConfig,
69    ) -> Self {
70        Self {
71            threshold,
72            semaphore: Arc::new(Semaphore::new(max_concurrent_tasks)),
73            inner: Arc::new(WorkerPoolInner {
74                hot_code_counter,
75                aot_compiler: Mutex::new(AotCompiler::new(config)),
76            }),
77        }
78    }
79
80    /// Spawns a compilation task for the given bytecode with the specified specId.
81    ///
82    /// # Arguments
83    ///
84    /// * `spec_id` - The specification ID for the EVM.
85    /// * `code_hash` - The hash of the bytecode to be compiled.
86    /// * `bytecode` - The bytecode to be compiled.
87    ///
88    /// # Returns
89    ///
90    /// A `JoinHandle` to the spawned task, which resolves to a `Result` indicating success or
91    /// failure.
92    pub(crate) fn spwan(
93        &self,
94        spec_id: SpecId,
95        code_hash: B256,
96        bytecode: Bytes,
97    ) -> Result<(), Error> {
98        let threshold = self.threshold;
99        let semaphore = self.semaphore.clone();
100        let inner = self.inner.clone();
101        if !inner.hot_code_counter.primary {
102            return Ok(());
103        }
104        let runtime = get_runtime();
105        runtime.spawn(async move {
106            let _permit = semaphore.acquire().await.unwrap();
107            // Check if the bytecode is all zeros
108            if code_hash.is_zero() {
109                return Ok(());
110            }
111            let counter = &inner.hot_code_counter;
112
113            // Read the current count of the bytecode hash from the embedded database
114            let count = counter.load_hot_call_count(code_hash)?;
115            let new_count = count + 1;
116            if new_count == threshold {
117                let aot_compiler = inner.aot_compiler.lock().await;
118                match aot_compiler.compile(code_hash, bytecode, spec_id).await {
119                    Ok(_) => {
120                        #[cfg(feature = "tracing")]
121                        tracing::info!("Compiled bytecode hash: {:#x}", code_hash);
122                    }
123                    Err(err) => {
124                        #[cfg(feature = "tracing")]
125                        tracing::error!(
126                            "Failed to compile bytecode hash: {:#x}, error: {:#?}",
127                            code_hash,
128                            err
129                        );
130                        // Skip updating the count of contract calls to ensure that
131                        // "so file doesn't exist" means the count is less than the threshold.
132                        return Err(err);
133                    }
134                }
135            }
136            // Only write the new count to the database after compiling successfully
137            counter.write_hot_call_count(code_hash, new_count)
138        });
139
140        Ok(())
141    }
142}
