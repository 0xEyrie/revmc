3/// Returns the default path
4fn default_path() -> PathBuf {
5    let home_dir = std::env::var("HOME").unwrap_or_else(|_| ".".to_string());
6    let config_path = {
7        #[cfg(test)]
8        {
9            ".test".to_string()
10        }
11        #[cfg(not(test))]
12        {
13            std::env::var("WORKER_STORE").unwrap_or_else(|_| ".revmc".to_string())
14        }
15    };
16    PathBuf::from(home_dir).join(config_path)
17}
